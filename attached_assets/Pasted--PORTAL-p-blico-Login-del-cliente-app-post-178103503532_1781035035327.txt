// Login del cliente
app.post("/api/portal/login", async (req, res) => {
  const { email, password } = req.body;
  const [user] = await db.select().from(clientPortalUsers)
    .where(eq(clientPortalUsers.email, email.toLowerCase()));
  if (!user || !user.activo) {
    return res.status(401).json({ message: "Email o contraseña incorrectos" });
  }
  if (user.password !== hashPortalPassword(password)) {
    return res.status(401).json({ message: "Email o contraseña incorrectos" });
  }
  // Guardar sesión del portal separada de la del admin
  (req.session as any).portalUserId = user.id;
  res.json({ ok: true, nombre: user.nombre, email: user.email });
});

// Logout del cliente
app.post("/api/portal/logout", (req, res) => {
  delete (req.session as any).portalUserId;
  res.json({ ok: true });
});

// Verificar sesión portal
app.get("/api/portal/me", async (req, res) => {
  const portalUserId = (req.session as any).portalUserId;
  if (!portalUserId) return res.status(401).json({ message: "No autenticado" });
  const [user] = await db.select().from(clientPortalUsers)
    .where(eq(clientPortalUsers.id, portalUserId));
  if (!user || !user.activo) return res.status(401).json({ message: "No autenticado" });
  res.json({ nombre: user.nombre, email: user.email });
});

// Listar informes del cliente logueado
app.get("/api/portal/reports", async (req, res) => {
  const portalUserId = (req.session as any).portalUserId;
  if (!portalUserId) return res.status(401).json({ message: "No autenticado" });
  const reports = await db.select({
    id: clientReports.id,
    titulo: clientReports.titulo,
    descripcion: clientReports.descripcion,
    tipoEstudio: clientReports.tipoEstudio,
    fechaEstudio: clientReports.fechaEstudio,
    pdfNombre: clientReports.pdfNombre,
    creadoEn: clientReports.creadoEn,
  }).from(clientReports)
    .where(eq(clientReports.clientPortalUserId, portalUserId))
    .orderBy(desc(clientReports.creadoEn));
  res.json(reports);
});

// Ver PDF de un informe (solo para imprimir, sin descarga)
app.get("/api/portal/reports/:id/pdf", async (req, res) => {
  const portalUserId = (req.session as any).portalUserId;
  if (!portalUserId) return res.status(401).json({ message: "No autenticado" });
  const [report] = await db.select().from(clientReports)
    .where(
      and(
        eq(clientReports.id, req.params.id),
        eq(clientReports.clientPortalUserId, portalUserId)
      )
    );
  if (!report) return res.status(404).json({ message: "No encontrado" });
  const buffer = Buffer.from(report.pdfData, "base64");
  // Content-Disposition inline para mostrar en browser (no descargar)
  res.setHeader("Content-Type", "application/pdf");
  res.setHeader("Content-Disposition", `inline; filename="${report.pdfNombre}"`);
  res.setHeader("X-Content-Type-Options", "nosniff");
  res.send(buffer);
});

<script>
  const APP_URL = 'https://TU-APP.replit.app';
  // Reemplazar con la URL real de la app

  let currentPdfUrl = null;

  async function handleLogin() {
    const email = document.getElementById('login-email').value;
    const pass = document.getElementById('login-pass').value;
    const errEl = document.getElementById('error-msg');
    errEl.style.display = 'none';

    try {
      const res = await fetch(`${APP_URL}/api/portal/login`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email, password: pass }),
        credentials: 'include',
      });
      if (!res.ok) {
        errEl.style.display = 'block';
        return;
      }
      const data = await res.json();
      showPortal(data.nombre);
      loadReports();
    } catch(e) {
      errEl.textContent = 'Error de conexión. Intentá de nuevo.';
      errEl.style.display = 'block';
    }
  }

  async function handleLogout() {
    await fetch(`${APP_URL}/api/portal/logout`, {
      method: 'POST', credentials: 'include'
    });
    document.getElementById('portal-screen').style.display = 'none';
    document.getElementById('login-screen').style.display = 'flex';
  }

  function showPortal(nombre) {
    document.getElementById('login-screen').style.display = 'none';
    document.getElementById('portal-screen').style.display = 'block';
    document.getElementById('portal-subtitle').textContent =
      `Bienvenido/a, ${nombre}. Podés visualizar e imprimir tus informes.`;
  }

  async function loadReports() {
    const res = await fetch(`${APP_URL}/api/portal/reports`,
      { credentials: 'include' });
    const reports = await res.json();
    const container = document.getElementById('reports-list');

    if (!reports.length) {
      container.innerHTML = `
        <div class="empty-state">
          <i class="ti ti-file-off"></i>
          <p>Aún no tenés informes disponibles.</p>
          <p style="font-size:12px;margin-top:6px">
            Te notificaremos por email cuando tu informe esté listo.
          </p>
        </div>`;
      return;
    }

    container.innerHTML = reports.map(r => `
      <div class="report-card">
        <div class="report-info">
          ${r.tipoEstudio ? `<div class="report-tipo">${r.tipoEstudio}</div>` : ''}
          <h3>${r.titulo}</h3>
          <p>${r.fechaEstudio ? `Fecha del estudio: ${r.fechaEstudio}` : ''}
          ${r.descripcion ? ` · ${r.descripcion}` : ''}</p>
        </div>
        <button class="print-btn" onclick="openPdf('${r.id}', '${r.titulo}')">
          <i class="ti ti-printer"></i> Ver e imprimir
        </button>
      </div>
    `).join('');
  }
